let num=parseInt(prompt("Enter a Number to check if it is Perfect Square or Not"))/** O(1) */

let count_Of_Factor=0;/** O(1) */

for(let i=1;i<=num;i++)/** O(n) */

    if(num%i===0) count_Of_Factor++
}
