Js function That Will Add Or Multiply:-

Write a Js function that will add or multiply all the arguments being provided. to it.

Please note users can provide any number of arguments but at least 3.

The first argument will always be a character (+ or * ) denoting what operation to perform.

Solution:


            let operation=(operationToBePerformed,second,third,...args)=>{
                switch(operationToBePerformed){
                    case "+":
                        var result=0
                        args.forEach((arg)=>{
                            result+=arg
                        })
                       result=result+second+third;
                       return result
                       break;
                    case "-":
                         var result=0
                         args.forEach((arg)=>{
                             result-=arg
                         })
                        result=second-third-result;
                        return result
                        break;
                    case "*":
                         var result=1
                         args.forEach((arg)=>{
                             result*=arg
                         })
                        result=second*third*result;
                        return result
                        break;
                    case "/":
                         var result=1
                         args.forEach((arg)=>{
                             result/=arg
                         })
                        result=second/third/result;
                        return result
                        break;
                     default:
                        alert("invalid Operation i.e first Argument")              
                }
            }
